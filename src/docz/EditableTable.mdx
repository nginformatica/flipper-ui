## With controlled state

<Playground>

        () => {
        
            const handleAdd = (item) => {
                if (item.someTitle < 10) {
                    window.alert('error, value can be smaller than 10')
                }
            }

            return (
                <EditableTable
                    paginationInfo
                    noRowsExpand
                    onAddRow={(item) => Promise.resolve(handleAdd(item))}
                    columns={[
                        {
                            title: 'some title',
                            field: 'someTitle',
                            type: 'numeric'
                        }
                    ]}
                    data={[
                        {
                            someTitle: 0
                        }
                    ]}
                />
            )
    }}
</Playground>
