## ChipField Example

<Playground>
    { 
        () => {
            const [state, setState] = React.useState([{ value: 'alpha' }])
            
            const handleRemove = value =>
                setState(state.filter(item => item.value !== value))

            return (
                <ChipField
                    fullwidth
                    fullwidthInput
                    label='chips'
                    variant='outlined' 
                    onAdd={item => setState([ { value: item }, ...state])}
                    onDelete={handleRemove}
                    values={state}/>
            )
        }
    }
</Playground>
